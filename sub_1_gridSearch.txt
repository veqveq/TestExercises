class Result {

    /*
     * Complete the 'gridSearch' function below.
     *
     * The function is expected to return a STRING.
     * The function accepts following parameters:
     *  1. STRING_ARRAY G
     *  2. STRING_ARRAY P
     */

    public static String gridSearch(List<String> G, List<String> P) {
        int finalRow = G.size() - P.size();

        for (int currentRow = 0; currentRow <= finalRow; currentRow++) {
            String currentGridString = G.get(currentRow);
            int index = 0;
            while (index != -1) {
                index = currentGridString.indexOf(P.get(0), index);
                if (index != -1) {
                    for (int i = 1; i < P.size(); i++) {
                        int nextIndex = G.get(currentRow + i).indexOf(P.get(i), index);
                        if (nextIndex != index) break;
                        if (i == P.size() - 1) return "YES";
                    }
                    index += 1;
                }
            }
        }
        return "NO";
    }
}
